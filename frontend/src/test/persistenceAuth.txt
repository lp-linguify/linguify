function testAuthPersistence() {
  console.log('ðŸ”„ TESTING AUTH PERSISTENCE');
  console.log('---------------------------');
  console.log('1. Current auth state:', localStorage.getItem('auth_state') ? 'Present' : 'Missing');
  
  // Store a flag to check after reload
  sessionStorage.setItem('auth_persistence_test', 'running');
  
  if (sessionStorage.getItem('auth_persistence_test') === 'completed') {
    // After reload
    console.log('2. Page has been reloaded');
    console.log('3. Auth state after reload:', localStorage.getItem('auth_state') ? 'Present' : 'Missing');
    sessionStorage.removeItem('auth_persistence_test');
    console.log('âœ… TEST COMPLETE');
  } else {
    // Before reload
    console.log('2. Reloading page...');
    sessionStorage.setItem('auth_persistence_test', 'completed');
    setTimeout(() => window.location.reload(), 1000);
  }
}

// Start test
testAuthPersistence();
